## docker/docker-compose.yml
# Docker Compose configuration for PGSIMS
# Production-ready setup for pgsims.alshifalab.pk with Caddy reverse proxy
#
# This configuration runs the backend on port 8014 (internal) which is proxied by Caddy
# Caddy handles SSL/TLS termination and routes traffic from pgsims.alshifalab.pk to port 8014
#
# SECURITY WARNING: Set environment variables in .env file:
#   - SECRET_KEY: A secure Django secret key
#   - DB_PASSWORD: A strong database password
#   - DB_USER: Database username (default: sims_user)
#   - DB_NAME: Database name (default: sims_db)
#   - ALLOWED_HOSTS: pgsims.alshifalab.pk,34.16.82.13,localhost,127.0.0.1
#   - CSRF_TRUSTED_ORIGINS: https://pgsims.alshifalab.pk
#   - CORS_ALLOWED_ORIGINS: https://pgsims.alshifalab.pk
#
# NEVER commit .env to version control.

services:
  # PostgreSQL Database
  db:
    image: postgres:15-alpine
    container_name: pgsims_db
    environment:
      # WARNING: Set DB_PASSWORD in .env file - never use default in production
      POSTGRES_DB: ${DB_NAME:-sims_db}
      POSTGRES_USER: ${DB_USER:-sims_user}
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
    healthcheck:
      test: [ "CMD-SHELL", "pg_isready -U ${DB_USER:-sims_user}" ]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - sims_network

  # Redis Cache & Message Broker
  redis:
    image: redis:7-alpine
    container_name: pgsims_redis
    command: redis-server --appendonly yes
    volumes:
      - redis_data:/data
    healthcheck:
      test: [ "CMD", "redis-cli", "ping" ]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - sims_network

  # Django Backend Application (runs on port 8014 for Caddy proxy)
  backend:
    build:
      context: ../backend
      dockerfile: Dockerfile
    container_name: pgsims_backend
    command: >
      sh -c "python manage.py migrate --noinput &&
             python manage.py collectstatic --noinput &&
             gunicorn sims_project.wsgi:application --bind 0.0.0.0:8014 --workers 4 --timeout 60"
    environment:
      - DEBUG=${DEBUG:-False}
      - SECRET_KEY=${SECRET_KEY}
      - ALLOWED_HOSTS=${ALLOWED_HOSTS:-pgsims.alshifalab.pk,34.29.33.138,localhost,127.0.0.1}
      - DATABASE_URL=postgresql://${DB_USER:-sims_user}:${DB_PASSWORD}@db:5432/${DB_NAME:-sims_db}
      - REDIS_URL=redis://redis:6379/0
      - CELERY_BROKER_URL=redis://redis:6379/1
      - CELERY_RESULT_BACKEND=redis://redis:6379/1
      - LOGIN_RATE_LIMIT=${LOGIN_RATE_LIMIT:-30/min}
      # Reverse proxy settings for Caddy
      - USE_PROXY_HEADERS=${USE_PROXY_HEADERS:-True}
      - SECURE_SSL_REDIRECT=${SECURE_SSL_REDIRECT:-True}
      - SESSION_COOKIE_SECURE=${SESSION_COOKIE_SECURE:-True}
      - CSRF_COOKIE_SECURE=${CSRF_COOKIE_SECURE:-True}
      - CSRF_TRUSTED_ORIGINS=${CSRF_TRUSTED_ORIGINS:-https://pgsims.alshifalab.pk}
      - CORS_ALLOWED_ORIGINS=${CORS_ALLOWED_ORIGINS:-https://pgsims.alshifalab.pk}
      - EMAIL_BACKEND=${EMAIL_BACKEND:-django.core.mail.backends.console.EmailBackend}
      - EMAIL_HOST=${EMAIL_HOST:-}
      - EMAIL_PORT=${EMAIL_PORT:-587}
      - EMAIL_USE_TLS=${EMAIL_USE_TLS:-True}
      - EMAIL_HOST_USER=${EMAIL_HOST_USER:-}
      - EMAIL_HOST_PASSWORD=${EMAIL_HOST_PASSWORD:-}
    ports:
      - "127.0.0.1:8014:8014" # Exposed for Caddy reverse proxy
    volumes:
      - static_volume:/app/staticfiles
      - media_volume:/app/media
      - ../backend/logs:/app/logs
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy
    healthcheck:
      test: [ "CMD", "curl", "-f", "http://localhost:8014/healthz/" ]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
    restart: unless-stopped
    networks:
      - sims_network

  # Celery Worker
  worker:
    build:
      context: ../backend
      dockerfile: Dockerfile
    container_name: pgsims_worker
    command: celery -A sims_project worker -l info --concurrency=2
    environment:
      - DEBUG=${DEBUG:-False}
      - SECRET_KEY=${SECRET_KEY}
      - DATABASE_URL=postgresql://${DB_USER:-sims_user}:${DB_PASSWORD}@db:5432/${DB_NAME:-sims_db}
      - REDIS_URL=redis://redis:6379/0
      - CELERY_BROKER_URL=redis://redis:6379/1
      - CELERY_RESULT_BACKEND=redis://redis:6379/1
    volumes:
      - media_volume:/app/media
      - ../backend/logs:/app/logs
    depends_on:
      - db
      - redis
    healthcheck:
      disable: true
    restart: unless-stopped
    networks:
      - sims_network

  # Celery Beat (Scheduler)
  beat:
    build:
      context: ../backend
      dockerfile: Dockerfile
    container_name: pgsims_beat
    command: celery -A sims_project beat -l info --scheduler django_celery_beat.schedulers:DatabaseScheduler
    environment:
      - DEBUG=${DEBUG:-False}
      - SECRET_KEY=${SECRET_KEY}
      - DATABASE_URL=postgresql://${DB_USER:-sims_user}:${DB_PASSWORD}@db:5432/${DB_NAME:-sims_db}
      - REDIS_URL=redis://redis:6379/0
      - CELERY_BROKER_URL=redis://redis:6379/1
      - CELERY_RESULT_BACKEND=redis://redis:6379/1
    volumes:
      - ../backend/logs:/app/logs
    depends_on:
      - db
      - redis
    healthcheck:
      disable: true
    restart: unless-stopped
    networks:
      - sims_network

  # Next.js Frontend Application
  frontend:
    build:
      context: ../frontend
      dockerfile: Dockerfile
      args:
        - NEXT_PUBLIC_API_URL=${NEXT_PUBLIC_API_URL:-/api}
    container_name: pgsims_frontend
    environment:
      - NODE_ENV=production
      - NEXT_PUBLIC_API_URL=${NEXT_PUBLIC_API_URL:-/api}
      - INTERNAL_API_URL=http://backend:8014
      - PORT=3000
      - HOSTNAME=0.0.0.0
    ports:
      - "127.0.0.1:8082:3000" # Exposed for Caddy reverse proxy
    depends_on:
      - backend
    healthcheck:
      test: [ "CMD", "node", "-e", "require('http').get('http://localhost:3000', (r) => {process.exit(r.statusCode === 200 ? 0 : 1)})" ]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s
    restart: unless-stopped
    networks:
      - sims_network

volumes:
  postgres_data:
  redis_data:
  static_volume:
  media_volume:


networks:
  sims_network:
    driver: bridge
